<script>

const COLORS = ["#C084FC","#818CF8","#60A5FA","#F472B6","#A78BFA","#38BDF8"];

const PRI = {

  low:    {bg:"#EDE9FE",text:"#6D28D9"},

  medium: {bg:"#EFF6FF",text:"#1D4ED8"},

  high:   {bg:"#FDF2F8",text:"#9D174D"},

};

const ST_NEXT  = {todo:"in-progress","in-progress":"done",done:"todo"};

const ST_ICON  = {todo:"○","in-progress":"◑",done:"●"};

const ST_LBL   = {todo:"To do","in-progress":"In progress",done:"Done"};

const ST_STYLE = {

  todo:          {bg:"#F5F3FF",text:"#6D28D9"},

  "in-progress": {bg:"#EFF6FF",text:"#1D4ED8"},

  done:          {bg:"#FDF2F8",text:"#9D174D"},

};


const KEY = "taskflow-v1";

let state = load();

let active = state.projects[0]?.id || null;

let dragging = null;

let newProjColor = COLORS[0];


function load() {

  try { return JSON.parse(localStorage.getItem(KEY)) || {projects:[],tasks:[]}; }

  catch { return {projects:[],tasks:[]}; }

}

function save() { localStorage.setItem(KEY, JSON.stringify(state)); }

function genId() { return Math.random().toString(36).slice(2,9); }


function formatDue(str) {

  if (!str) return null;

  const d = new Date(str + "T00:00:00");

  const today = new Date(); today.setHours(0,0,0,0);

  const diff = Math.round((d - today) / 86400000);

  const label = diff < 0 ? Math.abs(diff)+"d overdue" : diff===0?"Today":diff===1?"Tomorrow":diff+"d";

  const color = diff < 0 ? "#9D174D" : diff<=1 ? "#854F0B" : "#4B5563";

  const bg    = diff < 0 ? "#FDF2F8" : diff<=1 ? "#FAEEDA" : "#F3F4F6";

  return {label,color,bg};

}


function render() {

  renderSidebar();

  renderMain();

}


function renderSidebar() {

  const list = document.getElementById("proj-list");

  list.innerHTML = state.projects.map(p => {

    const count = state.tasks.filter(t=>t.projectId===p.id).length;

    const cls = p.id===active?"proj-btn active":"proj-btn";

    return `<button class="${cls}" style="--pc:${p.color}" onclick="selectProject('${p.id}')">

      <span class="proj-dot"></span>

      <span class="proj-name">${esc(p.name)}</span>

      <span class="proj-count">${count}</span>

    </button>`;

  }).join("");

}


function renderMain() {

  const proj = state.projects.find(p=>p.id===active);

  if (!proj) {

    document.getElementById("proj-title").textContent = "Select a project";

    document.getElementById("pct-badge").style.display = "none";

    document.getElementById("content").innerHTML = `<div class="no-proj"><div style="font-size:36px">✦</div><p>Create a project in the sidebar to get started.</p></div>`;

    return;

  }

  const tasks = state.tasks.filter(t=>t.projectId===active).sort((a,b)=>a.order-b.order);

  const done  = tasks.filter(t=>t.status==="done").length;

  const pct   = tasks.length ? Math.round(done/tasks.length*100) : 0;


  document.getElementById("proj-title").innerHTML = `<span class="title-dot" style="--pc:${proj.color}"></span>${esc(proj.name)}`;

  const badge = document.getElementById("pct-badge");

  badge.style.display = "";

  badge.textContent = pct+"% done";


  const groups = ["in-progress","todo","done"].map(st => {

    const grp = tasks.filter(t=>t.status===st);

    if (!grp.length) return "";

    return `<div style="margin-bottom:18px">

      <div class="group-label">${ST_LBL[st]}</div>

      ${grp.map(t => {

        const due = formatDue(t.due);

        const dueBadge = due ? `<span class="due-badge" style="background:${due.bg};color:${due.color}">${due.label}</span>` : "";

        return `<div class="task-card" id="tc-${t.id}" draggable="true"

          ondragstart="onDragStart(event,'${t.id}')"

          ondragover="onDragOver(event,'${t.id}')"

          ondrop="onDrop(event,'${t.id}')"

          ondragend="onDragEnd()">

          <span class="drag-handle">⠿</span>

          <button class="status-btn s-${st==='in-progress'?'ip':st}" onclick="cycleStatus('${t.id}')">${ST_ICON[t.status]}</button>

          <span class="task-title${t.status==='done'?' done':''}" title="${esc(t.title)}">${esc(t.title)}</span>

          ${dueBadge}

          <span class="pri-badge" style="background:${PRI[t.priority].bg};color:${PRI[t.priority].text}">${t.priority}</span>

          <select class="status-sel" style="background:${ST_STYLE[t.status].bg};color:${ST_STYLE[t.status].text}" onchange="updateStatus('${t.id}',this.value)">

            <option value="todo"${t.status==="todo"?" selected":""}>To do</option>

            <option value="in-progress"${t.status==="in-progress"?" selected":""}>In progress</option>

            <option value="done"${t.status==="done"?" selected":""}>Done</option>

          </select>

          <button class="del-btn" onclick="deleteTask('${t.id}')">✕</button>

        </div>`;

      }).join("")}

    </div>`;

  }).join("");


  document.getElementById("content").innerHTML = `

    <div class="stats">

      <div class="stat-card sc-p"><div class="stat-val">${tasks.length}</div><div class="stat-lbl">Total</div></div>

      <div class="stat-card sc-b"><div class="stat-val">${tasks.filter(t=>t.status==="in-progress").length}</div><div class="stat-lbl">In progress</div></div>

      <div class="stat-card sc-k"><div class="stat-val">${done}</div><div class="stat-lbl">Done</div></div>

    </div>

    <div class="prog-track"><div class="prog-fill" style="width:${pct}%"></div></div>

    <div class="add-row">

      <input class="task-input" id="task-input" placeholder="Add a task…" onkeydown="if(event.key==='Enter')addTask()" />

      <input class="due-input" id="due-input" type="date" title="Due date" />

      <select class="pri-sel" id="pri-sel">

        <option value="low">Low</option>

        <option value="medium" selected>Medium</option>

        <option value="high">High</option>

      </select>

      <button class="add-btn" onclick="addTask()">Add</button>

    </div>

    ${groups || '<div class="empty">No tasks yet — add one above.</div>'}

  `;

}


function esc(s) { return String(s).replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;").replace(/"/g,"&quot;"); }


function selectProject(id) { active = id; render(); }


function showNewProj() {

  newProjColor = COLORS[0];

  document.getElementById("new-proj-area").innerHTML = `

    <div class="new-proj-form">

      <input id="np-input" placeholder="Project name…" onkeydown="if(event.key==='Enter')addProject()" autofocus />

      <div class="color-row">${COLORS.map(c=>`<div class="swatch${c===newProjColor?" sel":""}" style="background:${c}" onclick="pickColor('${c}',this)"></div>`).join("")}</div>

      <div class="form-btns">

        <button class="form-add" onclick="addProject()">Add</button>

        <button class="form-cancel" onclick="cancelNewProj()">Cancel</button>

      </div>

    </div>`;

  setTimeout(()=>document.getElementById("np-input")?.focus(),50);

}


function pickColor(c, el) {

  newProjColor = c;

  document.querySelectorAll(".swatch").forEach(s=>s.classList.remove("sel"));

  el.classList.add("sel");

}


function cancelNewProj() {

  document.getElementById("new-proj-area").innerHTML = `<button class="add-proj-btn" onclick="showNewProj()">+ New project</button>`;

}


function addProject() {

  const name = document.getElementById("np-input")?.value?.trim();

  if (!name) return;

  const id = genId();

  state.projects.push({id, name, color: newProjColor});

  save(); active = id;

  cancelNewProj();

  render();

}


function addTask() {

  const title = document.getElementById("task-input")?.value?.trim();

  if (!title) return;

  const due = document.getElementById("due-input")?.value || "";

  const priority = document.getElementById("pri-sel")?.value || "medium";

  const maxOrder = state.tasks.filter(t=>t.projectId===active).reduce((m,t)=>Math.max(m,t.order),-1)+1;

  state.tasks.push({id:genId(), projectId:active, title, priority, status:"todo", due, order:maxOrder});

  save(); render();

}


function cycleStatus(id) {

  const t = state.tasks.find(t=>t.id===id);

  if (t) { t.status = ST_NEXT[t.status]; save(); render(); }

}


function updateStatus(id, val) {

  const t = state.tasks.find(t=>t.id===id);

  if (t) { t.status = val; save(); render(); }

}


function deleteTask(id) {

  state.tasks = state.tasks.filter(t=>t.id!==id);

  save(); render();

}


function onDragStart(e, id) { dragging = id; e.dataTransfer.effectAllowed = "move"; setTimeout(()=>{ const el=document.getElementById("tc-"+id); if(el) el.classList.add("dragging"); },0); }

function onDragOver(e, id)  { e.preventDefault(); if (dragging!==id) { document.querySelectorAll(".task-card").forEach(el=>el.classList.remove("drag-over")); const el=document.getElementById("tc-"+id); if(el) el.classList.add("drag-over"); } }

function onDrop(e, targetId) {

  e.preventDefault();

  if (!dragging || dragging===targetId) { onDragEnd(); return; }

  const projTasks = state.tasks.filter(t=>t.projectId===active).sort((a,b)=>a.order-b.order);

  const si = projTasks.findIndex(t=>t.id===dragging);

  const ti = projTasks.findIndex(t=>t.id===targetId);

  const [moved] = projTasks.splice(si,1);

  projTasks.splice(ti,0,moved);

  projTasks.forEach((t,i)=>{ const x=state.tasks.find(x=>x.id===t.id); if(x) x.order=i; });

  dragging = null; save(); render();

}

function onDragEnd() { dragging=null; document.querySelectorAll(".task-card").forEach(el=>{ el.classList.remove("dragging","drag-over"); }); }


render();

</script>
